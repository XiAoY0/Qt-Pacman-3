```
cd path/to/your/project   # 导航到你的项目目录
git init                  # 初始化一个新的 Git 仓库
git add .                 # 添加所有文件到暂存区
git commit -m "Initial commit"  # 提交文件

# 连接到 GitHub 仓库
git remote add origin https://github.com/your_username/your_repository_name.git

# 推送代码到 GitHub
git branch -M main        # 将默认分支名设置为 main
git push -u origin main   # 推送代码到 GitHub 仓库的 main 分支
```
